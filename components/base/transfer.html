# element-ui 穿梭框控件定制

<template>
  <el-dialog
    :title="headerTitle"
    :visible="transfer"
    width="800px"
    custom-class="transfer-dialog"
    :showClose="showClose"
    :close-on-click-modal="showClose"
    :close-on-press-escape="showClose"
    left
  >
    <div class="transfer-body clearfix">
      <div class="block-left">
        <div class="block-title">{{listTitle}}</div>
        <div class="block-body">
          <el-input
            v-model="state"
            :placeholder="placeholder"
            prefix-icon="el-icon-search"
            style="width:100%"
            v-if="transType=='single' || transType=='multi'"
          ></el-input>
          <ul v-if="transType=='single' && lists.length" class="left-single-ul">
            <li
              v-for="(item,index) in lists"
              :key="index"
              @click="click_li(index)"
              :class="{'checked':selected.length&&item.name == selected[0].name}"
            >
              <span>{{item.name}}</span>
            </li>
          </ul>
          <div
            v-else-if="transType=='multi' && lists.length>0"
            style="height:calc(100% - 55px);overflow-y:auto;margin-top:15px"
          >
            <el-checkbox
              :indeterminate="isIndeterminate"
              v-model="checkAll"
              @change="handleCheckAllChange"
            >全选</el-checkbox>
            <el-checkbox-group v-model="selected">
              <el-checkbox
                v-for="(item,index) in lists"
                :label="item"
                :key="index"
                @change="handleCheckedChange"
              >{{item.name}}</el-checkbox>
            </el-checkbox-group>
          </div>
          <div
            v-else-if="transType=='tree' && lists.length>0"
            style="height:calc(100% - 55px);overflow-y:auto;margin-top:15px"
          >
            <el-checkbox
              :indeterminate="isIndeterminate"
              v-model="checkAll"
              @change="handleCheckAllChange"
            >全选</el-checkbox>
          </div>
          <div v-else class="tac">
            <span class="inline-block mt25">暂无数据</span>
          </div>
        </div>
      </div>
      <div class="block-right">
        <div class="block-title">{{'已选择'}}</div>
        <div class="block-body">
          <ul>
            <li v-for="(item,index) in selected" :key="index">
              <span>{{item.name}}</span>
              <i class="el-icon-close" @click="unchecked(index)"></i>
            </li>
          </ul>
        </div>
      </div>
    </div>
    <span slot="footer" class="dialog-footer" style="text-align:right;">
      <el-button @click="close()">取消</el-button>
      <el-button :type="buttonType" @click="select_data()" :loading="loading">确定</el-button>
    </span>
  </el-dialog>
</template>
<script>
export default {
  name: 'BasicTransfer',
  props: {
    headerTitle: {
      type: String,
      default: '选择数据'
    },
    listTitle: {
      type: String,
      default: '数据列表'
    },
    listData: {
      type: Array,
      default: () => []
    },
    transType: {
      type: String,
      default: 'single'
    },
    buttonType: {
      type: String,
      default: 'primary'
    },
    placeholder: {
      type: String,
      default: '请输入关键字'
    },
    showClose: {
      type: Boolean,
      default: false
    },
    selectData: {
      type: Array,
      default: () => []
    },
    transfer: {
      type: Boolean,
      default: false
    },
    searchAsync: {
      type: Function,
      default: null
    },
    loading: {
      type: Boolean,
      default: false
    },
    searchKey: {
      type: String,
      default: 'id'
    }
  },
  data() {
    return {
      dia_trans: false,
      countNum: 0,
      state: '',
      timeout: null,
      lists: [...this.listData],
      selected: [],
      checkedData: [],
      selected_ids: []
    }
  },
  computed: {
    isIndeterminate() {
      if (this.transType !== 'multi') {
        return false
      }
      //MARK:如果存在着selectedData不包含在listData里的话，就把这个闭包去掉
      if (this.state == '') {
        if (
          this.selected.length > 0 &&
          this.lists.length > 0 &&
          this.selected.length != this.lists.length
        ) {
          return true
        }
        return false
      }

      if (this.interSection.length == 0) {
        return false
      }
      if (
        this.interSection.length != this.lists.length &&
        this.interSection.length > 0
      ) {
        return true
      }
      return false
    },
    checkAll: {
      get: function() {
        if (this.transType !== 'multi') {
          return false
        }
        //MARK:如果存在着selectedData不包含在listData里的话，就把这个闭包去掉
        if (this.state == '') {
          if (
            this.selected.length > 0 &&
            this.selected.length == this.lists.length
          ) {
            return true
          }
          return false
        }
        if (this.interSection.length == 0) {
          return false
        }
        if (
          this.interSection.length == this.lists.length &&
          this.interSection.length > 0
        ) {
          return true
        }
        return false
      },
      set: function() {}
    },
    interSection() {
      //MARK:如果存在着selectedData不包含在listData里的话，就把这个闭包去掉
      if (this.state == '') {
        return []
      }
      let _arr2Set = new Set(this.selected)
      return this.lists.filter(item => _arr2Set.has(item))
    }
  },
  watch: {
    state(newVal, oldVal) {
      if (newVal == '') {
        this.lists = [...this.listData]
      } else {
        this.lists = this.listData.filter(
          item => item.name.indexOf(newVal) != -1
        )
      }
    },
    selectData(newVal, oldVal) {
      this.selected = [...newVal]
    },
    listData(newVal, oldVal) {
      this.lists = [...newVal]
    }
  },
  methods: {
    handleCheckAllChange(val) {
      let all = [...this.lists]
      if (val == true) {
        let _arr2Set = new Set(this.selected)
        let intersection = all.filter(item => !_arr2Set.has(item))
        this.selected = this.selected.concat(intersection)
      } else {
        var map = new Map()
        this.selected.forEach(item => {
          map.get(item[this.searchKey]) == undefined &&
            map.set(item[this.searchKey], item)
        })
        all.forEach(item => {
          if (map.get(item[this.searchKey]) != undefined) {
            map.delete(item[this.searchKey])
          }
        })
        this.selected = [...map.values()]
      }
      //this.selected = this.array_unique_two_dimensional(this.selected)
    },
    handleCheckedChange() {
      // let _arr2Set = new Set(this.selected)
      // let intersection = this.lists.filter(item => _arr2Set.has(item))
      // this.checkAll = (intersection.length == this.lists.length&& this.lists.length>0 && intersection.length>0)?true:false
      // this.isIndeterminate = intersection.length == this.lists.length || this.lists.length == 0 || intersection.length == 0? false:true
      //this.selected = this.array_unique_two_dimensional(this.selected)
    },
    click_li(index) {
      this.selected = []
      this.selected.push(this.listData[index])
    },
    unchecked(index) {
      this.selected.splice(index, 1)
    },
    select_data() {
      this.checkedData = [...this.selected]
      this.selected_ids = []
      this.state = ''
      this.$emit('getSelectedData', this.checkedData, 'submit')
    },
    close() {
      this.checkedData = []
      this.selected = [...this.selectData]
      this.selected_ids = []
      this.state = ''
      this.$emit('getSelectedData', this.checkedData, 'close')
    },
    array_unique_two_dimensional(arr, param = 'id') {
      //数组去重
      const res = new Map()
      return arr.filter(
        a => !res.has(a[this.searchKey]) && res.set(a[this.searchKey], 1)
      )
    }
  },
  mounted() {}
}
</script>
