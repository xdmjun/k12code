    # 递归遍历数据

    // 删除指定ID的节点及子孙节点
changeTree(list, id) {
  let arr = [];
  if (list.length !== 0) {
    list.forEach(item => {
      if (item.id !== id) {
        arr.push(item);
        if (item.child.length >= 1) {
          item.child = this.changeTree(item.child, id);
        }
      }
    });
  }
  return arr;
}
